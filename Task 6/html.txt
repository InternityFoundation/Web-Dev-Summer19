function showDetails(subjects) {
  var subjectsType = subjects.getAttribute("data-subjects-type");
  alert("The " + subjects.innerHTML + " is related to " + subjectsType + ".");
}
function allowDrop(ev) {
  ev.preventDefault();
}

function drag(ev) {
  ev.dataTransfer.setData("Text", ev.target.id);
}

function drop(ev) {
  var data = ev.dataTransfer.getData("Text");
  ev.target.appendChild(document.getElementById(data));
  ev.preventDefault();
}
